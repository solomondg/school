\section*{Problem 26}
\subsection*{Solve for $a$ and $b$, then graph: }
\begin{equation*}
    \frac{d^2 x}{dt^2} - \frac{dx}{dt} - x = \cos{t}\\
\end{equation*}
\begin{equation*}
    x(t) = a \sin{t} + b \cos{t}
\end{equation*}
\subsection*{Solution:}
$$ x(t) = x''(t) - x'(t) - x(t)$$
$$x'(t) = a \cos{t} - b \sin{t}$$
$$x''(t) = - a \sin{t} + b \cos{t}$$
$$ (- a \sin{t} + b \cos{t}) - (a \cos{t} - b \sin{t}) - (a \sin{t} + b \cos{t}) = \cos{t}$$
$$-2a\sin{t} - a\cos{t} + b\sin{t} - 2b\cos{t}$$
$$(b-2a)\sin{t} + (-a-2b)\cos{t} = \cos{t}$$

\begin{equation*}
\sysdelim||\systeme{
    -2a + b = 0,
    -a - 2b = 1
}
\end{equation*}
\begin{equation*}
\sysdelim||\systeme{
    a = \frac{-1}{5},
    b = \frac{-2}{5}
}
\end{equation*}
